.model small    
.data

fmsg DW "Input :  $"
fmsg2 DW "Output Uppercase letter : $"
fmsg3 DW "Input is not Uppercase letter.$"


.code
main proc
    
    mov ax,@data
    mov ds,ax 
    
    lea dx,fmsg
    mov ah,9
    int 21h
    
    mov ah,1
    int 21h
    mov bh,al
    
    
    cmp bh,'A'
    jge bb
     
    call newline
    lea dx,fmsg3
    mov ah,9
    int 21h
    call newline
    jmp stop    
    
    bb:
    cmp bh,'Z'
    jle kk
    
    call newline
    lea dx,fmsg3
    mov ah,9
    int 21h
    call newline
    jmp stop
    
    kk:
     call newline
    lea dx,fmsg2
    mov ah,9
    int 21h 
    mov ah,2
    int 21h
    mov dl,bh
    int 21h
    call newline
    jmp stop 
    
    stop:
    
    jmp end_if
    
    
    end_if: 
    
    
    
    main endp

    Newline proc
        
    mov ah,2
    mov dl,0ah
    int 21h
    mov dl,0dh
    int 21h
    
    ret
    
    Newline endp
    
end main
